
Texture2D shaderTextures[16];
SamplerState SampleType;

struct PixelInputType
{
	float4 position : SV_POSITION;
	float2 tex : TEXCOORD0;
	float4 color : COLOR;
	int textureID : TEXTUREID;
};

float4 main(PixelInputType input) : SV_TARGET
{
	float4 textureColor;
	//TODO: if we dont do this we need to use Texture2DArray but then the textures all have to be equal size, this is probably easier when developing?
	if(input.textureID == 0)
		textureColor = shaderTextures[0].Sample(SampleType, input.tex);
	if(input.textureID == 1)
		textureColor = shaderTextures[1].Sample(SampleType, input.tex);
	
	textureColor = textureColor.zyxw;
//	textureColor = input.color;
	//textureColor = float4(0.0f, input.textureID, 0.0f, 1.0f);
	return textureColor;
}
